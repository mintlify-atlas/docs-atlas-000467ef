---
title: "Remix integration"
description: "Add Keystatic to an existing Remix v2 project and manage content through the Admin UI."
---

<Note>
  This guide assumes you have an existing Remix project. If you don't, run `npx create-remix@latest` to scaffold one first.
</Note>

<Steps>
  <Step title="Install dependencies">
    Install `@keystatic/core`, `@keystatic/remix`, and `@markdoc/markdoc`:

    <CodeGroup>

    ```bash npm
    npm install @keystatic/core @keystatic/remix @markdoc/markdoc
    ```

    ```bash pnpm
    pnpm add @keystatic/core @keystatic/remix @markdoc/markdoc
    ```

    ```bash yarn
    yarn add @keystatic/core @keystatic/remix @markdoc/markdoc
    ```

    </CodeGroup>
  </Step>

  <Step title="Create the Keystatic config file">
    Create `keystatic.config.ts` at the root of your project. This file defines your storage mode and content schema.

    The example below configures local storage and a single `posts` collection:

    ```typescript keystatic.config.ts
    import { config, fields, collection } from '@keystatic/core';

    export default config({
      storage: {
        kind: 'local',
      },
      collections: {
        posts: collection({
          label: 'Posts',
          slugField: 'title',
          path: 'app/content/posts/*',
          format: { contentField: 'content' },
          schema: {
            title: fields.slug({ name: { label: 'Title' } }),
            content: fields.markdoc({ label: 'Content' }),
          },
        }),
      },
    });
    ```

    <Note>
      `storage: { kind: 'local' }` is the recommended starting point. Content is read from and written to your local filesystem. When you're ready to deploy for a team, switch to [GitHub mode](/storage/github-mode).
    </Note>
  </Step>

  <Step title="Add the API route">
    Create `app/routes/api.keystatic.$.tsx`. This route handles all backend requests from the Admin UI:

    ```typescript app/routes/api.keystatic.$.tsx
    import type { ActionFunction, LoaderFunction } from '@remix-run/node';
    import { handleLoader } from '@keystatic/remix/api';
    import config from '../../keystatic.config';

    export const loader: LoaderFunction = args => handleLoader({ config }, args);
    export const action: ActionFunction = args => handleLoader({ config }, args);
    ```
  </Step>

  <Step title="Add the Admin UI route">
    Create `app/routes/keystatic.$.tsx`. This route renders the Keystatic Admin UI:

    ```typescript app/routes/keystatic.$.tsx
    import { makePage } from '@keystatic/remix/ui';
    import config from '../../keystatic.config';

    export default makePage(config);
    ```
  </Step>

  <Step title="Configure server dependencies">
    Keystatic is ESM-only, so it must be bundled into your server output. The configuration depends on whether you are using Vite or the traditional Remix compiler.

    <Tabs>
      <Tab title="Remix + Vite">
        Add `ssr.noExternal` to your Vite config:

        ```javascript vite.config.js
        import { defineConfig } from 'vite';
        import { remix } from '@remix-run/dev';

        export default defineConfig({
          plugins: [remix()],
          ssr: {
            noExternal: [/^@keystatic\//, 'minimatch'],
          },
        });
        ```
      </Tab>
      <Tab title="Traditional Remix (remix.config.js)">
        Add `serverDependenciesToBundle` to your `remix.config.js`:

        ```javascript remix.config.js
        /** @type {import('@remix-run/dev').AppConfig} */
        export default {
          ignoredRouteFiles: ['**/.*'],
          serverDependenciesToBundle: [/^@keystatic\//, 'minimatch'],
        };
        ```
      </Tab>
    </Tabs>
  </Step>

  <Step title="Start the dev server">
    <CodeGroup>

    ```bash npm
    npm run dev
    ```

    ```bash pnpm
    pnpm dev
    ```

    ```bash yarn
    yarn dev
    ```

    </CodeGroup>

    Visit `http://localhost:5173/keystatic` (or whichever port Remix uses) to open the Keystatic Admin UI.
  </Step>

  <Step title="Create your first post">
    In the Admin UI, navigate to **Posts** and create a new entry. Give it a title and some content, then click **Save**.

    Keystatic writes the file to `app/content/posts/` as a `.mdoc` file:

    ```
    app
    └── content
        └── posts
            └── my-first-post.mdoc
    ```

    The saved file will look like this:

    ```markdown app/content/posts/my-first-post.mdoc
    ---
    title: My First Post
    ---

    This is my very first post. I am **super** excited.
    ```
  </Step>

  <Step title="Read content in your Remix loaders">
    Keystatic provides a [Reader API](/reader/overview) for fetching content server-side. Use `createReader` inside Remix loader functions — it reads directly from the filesystem with no network requests.

    **List all posts**

    ```tsx app/routes/posts._index.tsx
    import { createReader } from '@keystatic/core/reader';
    import { Link, useLoaderData } from '@remix-run/react';
    import { json } from '@remix-run/node';
    import keystaticConfig from '../../keystatic.config';

    export async function loader() {
      const reader = createReader(process.cwd(), keystaticConfig);
      const posts = await reader.collections.posts.all();
      return json({ posts });
    }

    export default function PostsPage() {
      const { posts } = useLoaderData<typeof loader>();
      return (
        <ul>
          {posts.map(post => (
            <li key={post.slug}>
              <Link to={`/posts/${post.slug}`}>{post.entry.title}</Link>
            </li>
          ))}
        </ul>
      );
    }
    ```

    **Display a single post**

    ```tsx app/routes/posts.$slug.tsx
    import React from 'react';
    import Markdoc from '@markdoc/markdoc';
    import { createReader } from '@keystatic/core/reader';
    import { type LoaderFunctionArgs, json } from '@remix-run/node';
    import { useLoaderData } from '@remix-run/react';
    import keystaticConfig from '../../keystatic.config';

    export async function loader({ params }: LoaderFunctionArgs) {
      const reader = createReader(process.cwd(), keystaticConfig);
      const slug = params.slug;
      if (!slug) throw json('Not Found', { status: 404 });

      const post = await reader.collections.posts.read(
        slug,
        { resolveLinkedFiles: true }
      );
      if (!post) throw json('Not Found', { status: 404 });

      const errors = Markdoc.validate(post.content.node);
      if (errors.length) throw new Error('Invalid content');

      const content = Markdoc.transform(post.content.node);
      return json({ post: { title: post.title, content } });
    }

    export default function PostPage() {
      const { post } = useLoaderData<typeof loader>();
      return (
        <>
          <h1>{post.title}</h1>
          {Markdoc.renderers.react(post.content, React)}
        </>
      );
    }
    ```

    <Note>
      The `resolveLinkedFiles: true` option tells the reader to return the content node directly (as a parsed Markdoc AST) instead of an async function, which is more convenient in Remix loaders.
    </Note>
  </Step>
</Steps>

## Deployment

For team-based content editing on your deployed site, connect Keystatic to a GitHub repository using [GitHub mode](/storage/github-mode).
